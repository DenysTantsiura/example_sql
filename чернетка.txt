SELECT
  g.id, 
  g.group_name AS Group,
  sub.subject AS Subject,
  AVG(a.value_) AS Average
FROM
  assessments AS a
  JOIN
  subjects AS sub
  ON a.subject_id = sub.id
  JOIN
  students AS s
  ON a.student_id = s.id
  JOIN
  groups AS g
  ON s.group_id = g.id
--WHERE Subject = 'History'
GROUP BY
  g.id,
  Subject

  sub.subject AS Subject

  assessments AS a
  JOIN
  subjects AS sub
  ON a.subject_id = sub.id
  JOIN
  students AS s

  JOIN
  groups AS g
  ON s.group_id = g.id
